### 4.1 ModelValidator

**ファイル**: `services/validator.py`

**クラス構成**
```python
class ModelValidator:
    def __init__(self, model_data: PyMCModel):
        self.model_data = model_data
        self.errors: List[Dict[str, Any]] = []
        self.warnings: List[Dict[str, Any]] = []

    def validate(self) -> Dict[str, Any]:
        """全体バリデーション"""
        self._check_cycles()
        self._check_node_validity()
        self._check_dimension_compatibility()

        return {
            "valid": len(self.errors) == 0,
            "errors": self.errors,
            "warnings": self.warnings
        }
```

**バリデーション項目**

1. **循環依存チェック** (`_check_cycles`)
   - アルゴリズム: DFS（深さ優先探索）
   - エラー時: "Cyclic dependency detected"

2. **ノード妥当性チェック** (`_check_node_validity`)
   - 分布の存在確認
   - 必須パラメータの確認
   - パラメータ型の確認
   - 制約違反のチェック

3. **次元整合性チェック** (`_check_dimension_compatibility`)
   - トポロジカルソートの実行
   - 形状の伝播チェック

**制約チェック実装**
```python
def _check_constraint(self, node, param: str, value, constraint: str, param_type: str):
    """パラメータの制約をチェック"""
    # 配列の場合はスキップ
    if isinstance(value, list):
        return

    # 文字列を数値に変換
    try:
        if param_type == "int":
            num_value = int(value)
        elif param_type == "float":
            num_value = float(value)
        else:
            return
    except (ValueError, TypeError):
        return

    # 制約評価
    constraint_violated = False
    if constraint == "> 0":
        constraint_violated = not (num_value > 0)
    elif constraint == ">= 0":
        constraint_violated = not (num_value >= 0)
    # ... その他の制約

    if constraint_violated:
        self.errors.append({
            "node_id": node.id,
            "type": "constraint_violation",
            "message": f"Parameter '{param}' = {num_value} violates constraint: {constraint}"
        })
```

### 4.2 PyMCCodeGenerator

**ファイル**: `services/code_generator.py`

**クラス構成**
```python
class PyMCCodeGenerator:
    def __init__(self, model_data: PyMCModel):
        self.model_data = model_data
        self.indent = "    "

    def generate(self) -> str:
        """PyMCコード生成"""
        lines = []
        lines.extend(self._generate_imports())
        lines.append("")

        if self.model_data.csv_metadata:
            lines.extend(self._generate_data_loading())
            lines.append("")

        lines.append("with pm.Model() as model:")
        sorted_nodes = topological_sort(self.model_data)

        for node in sorted_nodes:
            if node.type == "data":
                continue
            node_code = self._generate_node_code(node)
            if node_code:
                lines.append(f"{self.indent}{node_code}")

        return "\n".join(lines)
```

**コード生成ロジック**

1. **インポート文生成** (`_generate_imports`)
   - PyMC、NumPy、Pandas（必要時）

2. **データ読み込み生成** (`_generate_data_loading`)
   - CSV読み込みコード
   - 観測データの抽出

3. **ノードコード生成** (`_generate_node_code`)
   - Computedノード: Python式として生成
   - Variableノード: pm.Distribution呼び出し
   - Featureデータ: pm.Data呼び出し

**パラメータ文字列構築**
```python
def _build_param_strings(self, node: Node, dist_name: str) -> List[str]:
    """パラメータ文字列のリストを構築"""
    param_strs = []
    for param, value in node.parameters.items():
        if isinstance(value, str):
            if value.startswith("@"):
                # 参照: @を除去
                param_strs.append(f"{param}={value[1:]}")
            else:
                # 値: そのまま
                param_strs.append(f"{param}={value}")
        elif isinstance(value, list):
            # 配列: リストとして出力
            param_strs.append(f"{param}={value}")
        else:
            param_strs.append(f"{param}={value}")
    return param_strs
```

class InferenceEngine:
    def __init__(self, model_data: PyMCModel):
        self.model_data = model_data
        self.code_generator = PyMCCodeGenerator(model_data)

    def run_inference(self, config: InferenceConfig) -> Dict[str, Any]:

        summary = self._generate_summary(trace, config)

        return {
            "success": True,
            "execution_time": elapsed_time,
            "trace_plot": plots['trace'],
            "posterior_plot": plots['posterior'],
            "summary": summary
        }

**MCMC実行**
```python
def _run_mcmc(self, model: pm.Model, config: InferenceConfig) -> az.InferenceData:
    """MCMC sampling"""
    with model:
        if config.sampler == "NUTS":
            trace = pm.sample(
                draws=config.draws,
                tune=config.tune,
                chains=config.chains,
                cores=config.cores,
                return_inferencedata=True
            )
        else:
            trace = pm.sample(
                draws=config.draws,
                step=pm.Metropolis(),
                tune=config.tune,
                chains=config.chains,
                cores=config.cores,
                return_inferencedata=True
            )
    return trace
```

**VI実行**
```python
def _run_vi(self, model: pm.Model, config: InferenceConfig) -> az.InferenceData:
    """Variational Inference"""
    with model:
        if config.vi_method == "ADVI":
            approx = pm.fit(n=config.iterations, method='advi')
        else:
            approx = pm.fit(n=config.iterations, method='fullrank_advi')

        trace = approx.sample(draws=config.draws)
    return trace
```

def _generate_plots(self, trace: az.InferenceData, config: InferenceConfig) -> Dict[str, str]:

## 5. ユーティリティ層設計

### 5.1 model_utils.py

**責務**: モデルに関する汎用ロジック

**主要関数**

1. **topological_sort**: トポロジカルソート
   - アルゴリズム: Kahn's algorithm
   - ノードを依存順に並び替え

2. **infer_shape**: 配列の形状推論
   - ネストした配列から次元を抽出

3. **infer_shape_from_parameters**: パラメータから形状推論
   - 全パラメータの形状を解析
   - 最大次元の形状を返却

```python
def topological_sort(model_data: PyMCModel) -> List[Node]:
    """トポロジカルソート (Kahn's algorithm)"""
    in_degree = {node.id: 0 for node in model_data.nodes}
    for edge in model_data.edges:
        in_degree[edge.target] += 1

    queue = [node for node in model_data.nodes if in_degree[node.id] == 0]
    sorted_nodes = []

    while queue:
        node = queue.pop(0)
        sorted_nodes.append(node)

        for edge in model_data.edges:
            if edge.source == node.id:
                in_degree[edge.target] -= 1
                if in_degree[edge.target] == 0:
                    target_node = next(n for n in model_data.nodes if n.id == edge.target)
                    queue.append(target_node)

    return sorted_nodes
```

**責務**: 実行可能なPyMCモデルの構築

```python
def build_pymc_model(model_data: PyMCModel) -> pm.Model:
    """実際のPyMCモデルを構築して返す（推論実行用）"""
    df = load_csv_data(model_data)
    data_node_map, data_node_roles = build_data_node_map(model_data)

    model = pm.Model()

    with model:
        sorted_nodes = topological_sort(model_data)
        node_vars = {}

        for node in sorted_nodes:
            if node.type == "data":
                continue

            var_name = node.name

            # 決定論的ノード
            if node.type == "computed":
                if not node.expression:
                    continue
                expression = _resolve_expression_for_exec(node.expression, node_vars)
                node_vars[var_name] = expression
                continue

            # Featureデータ
            if node.observed and hasattr(node, 'observed_data_source') and node.observed_data_source:
                column_name = data_node_map.get(node.observed_data_source)
                role = data_node_roles.get(node.observed_data_source, 'unused')

                if column_name and df is not None and column_name in df.columns:
                    data_values = df[column_name].values

                    if role == 'feature':
                        node_vars[var_name] = pm.Data(var_name, data_values)
                        continue

            # 確率変数
            if not node.distribution:
                continue

            dist_name = node.distribution
            params = _build_distribution_params(node, model_data, node_vars)
            shape = tuple(node.shape) if node.shape else None
            observed = _get_observed_data(node, data_node_map, data_node_roles, df)

            dist_class = getattr(pm, dist_name)
            node_vars[var_name] = dist_class(var_name, shape=shape, observed=observed, **params)

    return model
```

    """式を評価して実行可能な形式に変換"""
